\newcommand{\attributes}[1]
{
  \attribute {
    \centering
    \begin{tabular}{@{}lll@{}}
      #1
    \end{tabular}
    }
}

\begin{document}
\begin{tikzpicture}
  \ttfamily

  \begin{object}[text width=20em]{Employee}{0,-1em}
    \attributes
    {
      PK & Id	         & uniqueidentifier \\
         & Name        & nvarchar(max) \\
         & Email       & nvarchar(max) \\
         & ProviderId  & nvarchar(30)
    }
  \end{object}


  \begin{object}[text width=20em]{Job}{60em,-1em}
    \attributes
    {
      PK  & Id	         & uniqueidentifier \\
          & Title        & nvarchar(100)    \\
          & Longitude    & decimal          \\
          & Latitude     & decimal          \\
          & Radius       & decimal
    }
  \end{object}

  \begin{object}[text width=20em]{Clock}{30em,0}
    \attributes
    {
      PK  & Id         & uniqueidentifier \\
      FK  & JobId      & uniqueidentifier \\ 
      FK  & EmployeeId & uniqueidentifier \\
      FK? & ParentId   & uniqueidentifier \\
          & Type       & nvarchar(3) \\
          & ExpectedAt & datetime2 \\
      ?   & ActualAt   & datetime2
    }
  \end{object}

  \association{Employee}{1}{}{Clock}{0..*}{}
  \association{Job}{1}{}{Clock}{0..*}{}

  % clock self association
  \draw[-]
  (Clock.east) --
  ++(3em, 0) -- 
  ++(0, -8em) -- 
  (30em, -14.25em) -- % not janky at all
  (Clock.south);
  \node [below] at (Clock.south) {0..1};

\end{tikzpicture}
